    <script>
        const countdownEl = document.getElementById('countdown');
        const mainContent = document.getElementById('main-content');
        let count = 5;

        // Start Countdown
        const timerInterval = setInterval(() => {
            count--;
            if (count > 0) {
                countdownEl.innerText = count;
                // Retrigger animation
                countdownEl.style.animation = 'none';
                void countdownEl.offsetWidth; // Trigger reflow
                countdownEl.style.animation = 'popInOut 1s cubic-bezier(0.25, 0.1, 0.25, 1) forwards';
            } else {
                clearInterval(timerInterval);
                countdownEl.style.display = 'none';
                explodeFlowers();
            }
        }, 1000);

        function explodeFlowers() {
            const flowers = ['🌸', '🌺', '🌻', '🌹', '🌷', '🌼', '💐', '✨'];
            const numFlowers = window.innerWidth < 600 ? 50 : 80;

            for (let i = 0; i < numFlowers; i++) {
                let flower = document.createElement('div');
                flower.innerText = flowers[Math.floor(Math.random() * flowers.length)];
                flower.className = 'flower-particle';
                document.body.appendChild(flower);

                // Random angle and distance
                let angle = Math.random() * Math.PI * 2;
                let distance = Math.random() * (window.innerWidth / 1.5) + 50; 
                let tx = Math.cos(angle) * distance;
                let ty = Math.sin(angle) * distance;
                let rot = Math.random() * 360 + 'deg';

                flower.style.setProperty('--tx', `${tx}px`);
                flower.style.setProperty('--ty', `${ty}px`);
                flower.style.setProperty('--rot', rot);
                flower.style.animation = `blast ${Math.random() * 0.8 + 0.8}s cubic-bezier(0.1, 0.8, 0.3, 1) forwards`;

                // Remove particle after animation
                setTimeout(() => flower.remove(), 1600);
            }

            // Show main greeting shortly after blast begins
            setTimeout(showGreeting, 800);
        }

        function showGreeting() {
            mainContent.style.display = 'flex';
            mainContent.style.animation = 'fadeInScale 1.5s cubic-bezier(0.25, 1, 0.5, 1) forwards';
            
            // Start the lovely floating animations
            setInterval(spawnFloatingLove, 300);
            
            // Initial burst of love
            for(let i=0; i<10; i++) {
                setTimeout(spawnFloatingLove, i * 100);
            }
        }

        function spawnFloatingLove() {
            const loveEmojis = ['💖', '✨', '🥰', '🎀', '💕', '🧸', '💋', '🍓', '💌', '🌸'];
            let heart = document.createElement('div');
            heart.innerText = loveEmojis[Math.floor(Math.random() * loveEmojis.length)];
            heart.className = 'floating-heart';
            
            // Random horizontal start
            heart.style.left = Math.random() * 100 + 'vw';
            
            // Random animation variables
            let duration = Math.random() * 4 + 4; // 4s to 8s
            let drift = (Math.random() * 200 - 100) + 'px';
            let rot = (Math.random() * 360) + 'deg';
            
            heart.style.setProperty('--drift', drift);
            heart.style.setProperty('--rot', rot);
            heart.style.animationDuration = duration + 's';
            
            // Random sizing for depth
            heart.style.transform = `scale(${Math.random() * 0.8 + 0.5})`;
            
            document.body.appendChild(heart);
            
            // Clean up
            setTimeout(() => heart.remove(), duration * 1000);
        }
